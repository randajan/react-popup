## Install

```bash
npm install --save @randajan/react-popup
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from '@randajan/react-popup'
import '@randajan/react-popup/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
